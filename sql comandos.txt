USE valorantav3;
-- Criar a tabela Usu√°rio
CREATE TABLE Usuario (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    VP INT,
    LEVELXP INT,
    ELO VARCHAR(65),
    NOME_USUARIO VARCHAR(255),
    DATA_CONTA DATE
);

-- Criar a tabela Agente
CREATE TABLE Agente (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    ROLE_AGENTE VARCHAR(255),
    NOME_AGENTE VARCHAR(255),
    ULTIMATE VARCHAR(255)
);

-- Criar a tabela Arma
CREATE TABLE Arma (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NOME_ARMA VARCHAR(255),
    TIPO_ARMA VARCHAR(255),
    SKIN VARCHAR(255)
);

-- Criar a tabela Loja
CREATE TABLE Loja (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    DATA_LOJA DATE,
    PRECO DECIMAL(10, 2),
    RARIDADE VARCHAR(255),
    TIPO_ARMA INT,
    FOREIGN KEY (TIPO_ARMA) REFERENCES Arma(ID)
);

-- Criar a tabela Partida
CREATE TABLE Partida (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    MODO VARCHAR(255),
    DATA_PARTIDA DATE,
    ID_USUARIO INT,
    MAPA VARCHAR(255),
    FOREIGN KEY (ID_USUARIO) REFERENCES Usuario(ID)
);
CREATE INDEX idx_nome_usuario ON Usuario (NOME_USUARIO);
CREATE INDEX idx_id_usuario ON Partida (ID_USUARIO);
